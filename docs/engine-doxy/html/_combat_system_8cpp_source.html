20 {
21  // Running away.
22  if(!run_away_queue_.empty())

24  run_away_from_(

26  std::get<1>(run_away_queue_.front()),
27  std::get<2>(run_away_queue_.front())
28  );
29  run_away_queue_.pop();
30  }
31 
32  // Autoattacks.
33  for(auto& ent : entities_.get_component_container<CombatComponent>())
34  {
35  if(ent.second.cd_time < ent.second.cooldown)
36  {
37  ent.second.cd_time += delta;
38  continue;
39  }
40  else
41  ent.second.cd_time = 0;
42 
43  if(ent.second.curr_target != Component::NO_ENTITY &&
44  entities_.exists(ent.second.curr_target))
45  {
46  auto phys_comp = entities_.get_component<PhysicsComponent>(ent.first);
47  auto target_phys_comp = entities_.get_component<PhysicsComponent>(ent.second.curr_target);
48 
49  auto range = ent.second.range * ent.second.range;
50  auto task_comp = entities_.get_component<TaskHandlerComponent>(ent.first);
51  if(phys_comp && target_phys_comp)
52  {

54  auto path_comp = entities_.get_component<PathfindingComponent>(ent.first);
55 
60  bool is_in_sight{in_sight(ent.first, ent.second.curr_target)};
61  bool kill_task_assigned{task_comp ? (TaskHelper::get_task_type(entities_, task_comp->curr_task) == TASK_TYPE::KILL) : false};
62  bool on_path{path_comp && !path_comp->path_queue.empty()};
63  bool is_in_range{dist <= range};
64  if((!is_in_range || !is_in_sight) && !on_path && kill_task_assigned)
65  {
66  if(ent.second.pursue)
67  { // Order it to go after the target again.
68  TaskHelper::add_task(entities_, ent.first, ent.second.curr_target, true);
69  task_comp->task_queue.push_front(task_comp->curr_task);
70  auto new_task = TaskHelper::create_task(entities_, ent.second.curr_target, TASK_TYPE::GET_IN_RANGE);
71  task_comp->task_queue.push_front(new_task);
72  task_comp->curr_task = Component::NO_ENTITY;
73  }
74  else
75  { // Just give up.
76  TaskHelper::cancel_task(entities_, task_comp->curr_task);
77  }
78 
79  ent.second.curr_target = Component::NO_ENTITY;
80  }
81  else if(is_in_range && is_in_sight)
82  {
83  if(on_path) // Enemy came to us, just start attacking!
84  path_comp->path_queue.clear();
85 
86  auto dmg = CombatHelper::get_dmg(ent.second.min_dmg, ent.second.max_dmg);
87  GraphicsHelper::look_at(entities_, ent.first, ent.second.curr_target);
88  switch(ent.second.atk_type)
89  {
90  case ATTACK_TYPE::MELEE:
91  HealthHelper::sub_health(entities_, ent.second.curr_target, dmg);
92  OnHitHelper::call(entities_, ent.second.curr_target, ent.first);
93  if(HealthHelper::get_health(entities_, ent.second.curr_target) <= 0)
94  {
95  DestructorHelper::destroy(entities_, ent.second.curr_target, false, ent.first);
96  ent.second.curr_target = Component::NO_ENTITY;
97  ent.second.cd_time = ent.second.cooldown; // Allows to attack again instantly.
98  if(task_comp && TaskHelper::get_task_type(entities_, task_comp->curr_task) == TASK_TYPE::KILL)
99  TaskHelper::cancel_task(entities_, task_comp->curr_task);
100  }
101  break;
102  case ATTACK_TYPE::RANGED:
103  create_homing_projectile(ent.first, ent.second);
104  break;
105  }
106  }
107  else if(!kill_task_assigned)
108  ent.second.curr_target = Component::NO_ENTITY;
109  }
110  else // Target killed or ran away.
111  ent.second.curr_target = Component::NO_ENTITY;
112  }
113  }
114 
115  // Homing projectiles.
116  for(auto& ent : entities_.get_component_container<HomingComponent>())
117  {
118  if(ent.second.target == Component::NO_ENTITY)
119  continue; // Manually spawned.
120  else if(!entities_.exists(ent.second.target)) // Target killed.
121  DestructorHelper::destroy(entities_, ent.first);
122 
123  auto mov_comp = entities_.get_component<MovementComponent>(ent.first);
124  auto phys_comp = entities_.get_component<PhysicsComponent>(ent.first);
125  auto graph_comp = entities_.get_component<GraphicsComponent>(ent.first);
126  auto enemy_graph_comp = entities_.get_component<GraphicsComponent>(ent.second.target);
127 
128  if(mov_comp && phys_comp && graph_comp && enemy_graph_comp &&
129  graph_comp->node && graph_comp->entity && enemy_graph_comp->entity)
130  { // Moves the homing projectile in the target's direction and checks if hit occured.
131  auto& pos = phys_comp->position;
132  auto enemy_pos = enemy_graph_comp->node->getPosition();
133  auto dir = enemy_pos - pos;
134  dir.normalise();
135 
136  pos += dir * mov_comp->speed_modifier;
137  graph_comp->node->setPosition(pos);
138 
139  if(graph_comp->entity->getWorldBoundingBox(true).intersects(enemy_graph_comp->entity->getWorldBoundingBox(true)))
140  { // That's a hit.
141  HealthHelper::sub_health(entities_, ent.second.target, ent.second.dmg);
142  OnHitHelper::call(entities_, ent.second.target, ent.second.source);
143  if(HealthHelper::get_health(entities_, ent.second.target) <= 0)
144  {
145  DestructorHelper::destroy(entities_, ent.second.target, false, ent.second.source);
146 
147  auto task_comp = entities_.get_component<TaskHandlerComponent>(ent.second.source);
148  if(task_comp && TaskHelper::get_task_type(entities_, task_comp->curr_task) == TASK_TYPE::KILL)
149  TaskHelper::cancel_task(entities_, task_comp->curr_task);
150  }
151  DestructorHelper::destroy(entities_, ent.first, false, ent.second.target);
152  }
153  }
154  }

301 void CombatSystem::run_away_from_(std::size_t id, std::size_t from_id, std::size_t min_node_count)
302 {
303  auto phys = entities_.get_component<PhysicsComponent>(id);
304  auto path = entities_.get_component<PathfindingComponent>(id);
305  if(phys && path)
306  {
307  std::size_t attempts{0};
308  std::size_t start_node{Grid::instance().get_node_from_position(phys->position.x, phys->position.z)};
309  std::size_t target{Component::NO_ENTITY}, target_node_count{0};
310 
311  std::deque<std::size_t> path{};
312  Ogre::Real min_distance{}; // Closest node from the path to the enemy.
313  util::heuristic::RUN_AWAY_HEURISTIC heuristic{entities_, from_id};
314 
315  while(attempts < max_run_away_attempts_)
316  { // Check different random nodes to find the furthest one.
317  std::size_t new_target{Grid::instance().get_random_free_node()};
318  path = util::DEFAULT_PATHFINDING_ALGORITHM::get_path(entities_, id, start_node, new_target, heuristic, false);
319 
320  Ogre::Real new_min_distance{std::numeric_limits<Ogre::Real>::max()};
321  for(const auto& node : path)
322  {
323  Ogre::Real dist = PhysicsHelper::get_distance(entities_, node, from_id);
324  new_min_distance = (dist < new_min_distance) ? dist : new_min_distance;
325  }
326 
327  if(new_min_distance > min_distance && path.size() > target_node_count
328  && path.size() > min_node_count)
329  {
330  target = new_target;
331  min_distance = new_min_distance;
332  }
333  ++attempts;
334  }
335 
336  if(!path.empty()) // Perform the actual pathfinding for the best target found.
337  util::pathfind<util::DEFAULT_PATHFINDING_ALGORITHM>(entities_, id, target, heuristic, true, false);
338  }
339 }
