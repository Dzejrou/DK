34  std::size_t id = Component::NO_ENTITY;
35  std::size_t queue_size = std::numeric_limits<std::size_t>::max();
36  Ogre::Real distance = std::numeric_limits<Ogre::Real>::max();
37  for(const auto& ent : ents.get_component_container<CommandComponent>())
38  {

40  continue;
41

43  if(comp)
44  {
45  Ogre::Real new_distance = PhysicsHelper::get_distance(ents, ent.first, target);
46  if(comp->task_queue.size() < queue_size || (comp->task_queue.size() == queue_size &&
47  new_distance < distance))
48  {
49  queue_size = comp->task_queue.size();

51  distance = new_distance;
52  }
53  }
54  }

56  {
