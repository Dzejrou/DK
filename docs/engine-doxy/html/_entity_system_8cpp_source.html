62 void EntitySystem::cleanup()
63 {
64  // Remove components.
65  for(auto& ent : components_to_be_removed_)
66  {
67  auto it = entities_.find(ent.first);
68  if(it != entities_.end())
69  {
70  it->second.set(ent.second, false);
71  if(!it->second.any()) // No components remaining -> destroy it.
72  to_be_destroyed_.push_back(ent.first);
73  }
74  delete_component_now(ent.first, ent.second);
75  }
76  components_to_be_removed_.clear();
77 
84  std::vector<tdt::uint> tmp_to_be_destroyed_{};
85  tmp_to_be_destroyed_.swap(to_be_destroyed_);
86  for(auto id : tmp_to_be_destroyed_)
87  {
88  if(id == Component::NO_ENTITY)
89  continue;

92  if(entity == entities_.end())
93  continue;

95  for(tdt::uint i = 0; i < entity->second.size(); ++i)
96  {
97  if(entity->second.test(i))
98  delete_component_now(id, i);
99  }
100  entities_.erase(id);
101  }

103  // Call constructors.

109 tdt::uint EntitySystem::create_entity(const std::string& table_name, const Ogre::Vector3& position)
110 {
111  tdt::uint id = get_new_id();
112  entities_.emplace(std::make_pair(id, std::bitset<Component::count>{}));
113 
114  if(table_name == "") // Allows to create empty entities that are setup manually.
115  return id;
116  else
117  register_entity(table_name);
118 
119  auto& bits = entities_.find(id)->second;
120 
121  lpp::Script& script = lpp::Script::instance();
122  std::vector<int> comps = script.get_vector<int>(table_name + ".components");
123 
124  for(auto component_type : comps)
125  {
126  if(component_type < 0 || component_type >= (int)bits.size())
127  continue; // Maybe notify in the console? Make the console a singleton?
128 
129  bits.set(component_type); // Duplicate components will just overwrite, no need for error checking.
130  if(component_type >= 0 && component_type < Component::count && loaders_[component_type])
131  LOAD_COMPONENT(component_type, id, table_name);
132  }
133 
134  if(bits.test(GraphicsComponent::type)) // Y coordinate already set as half height.
135  PhysicsHelper::set_2d_position(*this, id, Ogre::Vector2{position.x, position.z});
136  else
137  PhysicsHelper::set_position(*this, id, position);
138 
139  if(bits.test(ConstructorComponent::type))
140  constructors_to_be_called_.emplace_back(id);
141  return id;
142 }
