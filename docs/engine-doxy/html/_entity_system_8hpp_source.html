632 template<>
633 inline void EntitySystem::load_component<GraphicsComponent>(tdt::uint id, const std::string& table_name)
634 {
635  lpp::Script& script = lpp::Script::instance();
636  std::string mesh = script.get<std::string>(table_name + ".GraphicsComponent.mesh");
637  std::string material = script.get<std::string>(table_name + ".GraphicsComponent.material");
638  auto res = graphics_.emplace(id, GraphicsComponent{std::move(mesh), std::move(material)});
639 
640  // Ogre init of the entity and scene node.
641  auto& comp = res.first->second;
642  comp.node = scene_.getRootSceneNode()->createChildSceneNode("entity_" + std::to_string(id));
643  comp.entity = scene_.createEntity(comp.mesh);
644  comp.node->attachObject(comp.entity);
645  comp.entity->setQueryFlags(1);
646 
647  if(!script.get<bool>(table_name + ".GraphicsComponent.visible"))
648  {
649  comp.visible = false;
650  comp.node->setVisible(false);
651  }
652 
653  if(script.get<bool>(table_name + ".GraphicsComponent.manual_scaling"))
654  {
655  comp.manual_scaling = true;
656  tdt::real x, y, z;
657  x = script.get<tdt::real>(table_name + ".GraphicsComponent.scale_x");
658  y = script.get<tdt::real>(table_name + ".GraphicsComponent.scale_y");
659  z = script.get<tdt::real>(table_name + ".GraphicsComponent.scale_z");
660  comp.scale = Ogre::Vector3{x, y, z};
661  comp.node->setScale(comp.scale);
662  }
663 
664  if(comp.material != "NO_MAT")
665  comp.entity->setMaterialName(comp.material);
666 
667  // Make the entity stand on ground.
668  auto half_height = comp.entity->getWorldBoundingBox(true).getHalfSize().y;
669  auto phys_comp = get_component<PhysicsComponent>(id);
670  if(phys_comp)
671  {
672  phys_comp->half_height = half_height;
673  phys_comp->position = Ogre::Vector3{phys_comp->position.x, half_height, phys_comp->position.z};
674  comp.node->setPosition(phys_comp->position);
675  }
676 
677  // This will allow specific querying.
678  if(!script.is_nil(table_name + ".GraphicsComponent.query_flags"))
679  comp.entity->setQueryFlags(script.get<int>(table_name + ".GraphicsComponent.query_flags"));
680 
681  // Attach a light if a light component was loaded before the graphics one.
682  auto light = get_component<LightComponent>(id);
683  if(light && light->light && comp.node)
684  comp.node->attachObject(light->light);
685 
686  comp.entity->setRenderingDistance(4000.f);
687 }
