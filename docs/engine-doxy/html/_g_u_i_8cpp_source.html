414 bool GUI::escape_pressed()

422  };
423 
428  if(window_->getChild("MAIN_MENU")->isVisible() &&
429  !window_->getChild("MAIN_MENU/NEW_GAME_DIALOG")->isVisible() &&
430  !window_->getChild("OPTIONS")->isVisible() &&
431  game_->state_ != GAME_STATE::ENDED && game_->state_ != GAME_STATE::INTRO_MENU)
432  {
433  game_->set_state(GAME_STATE::RUNNING);
434  set_visible("MESSAGE", false);
435  window_->getChild("MAIN_MENU")->setVisible(false);
436  window_->getChild("TOOLS")->setVisible(true);
437  window_->getChild("ENTITY_VIEW")->setVisible(true);
438  window_->getChild("GAME_LOG")->setVisible(true);
439  window_->getChild("NEXT_WAVE")->setVisible(true);
440  }
441 
442  for(const auto& sub_window : sub_windows)
443  {
444  if(window_->getChild(sub_window)->isVisible())
445  {
446  window_->getChild(sub_window)->setVisible(false);
447  res = true;
448  }
449  }
450 
451  return res;
452 }

470 void GUI::list_directory(const std::string& dir, CEGUI::Listbox& box, bool strip_ext)
471 {
472  box.resetList();
473 
474 #ifdef WIN32
475  HANDLE handle;
476  WIN32_FIND_DATA fdata;
477 
478  handle = FindFirstFile(dir.c_str(), &fdata);
479  std::string tmp;
480  if(handle != INVALID_HANDLE_VALUE)
481  {
482  do
483  {
484  tmp = fdata.cFileName;
485  if(strip_ext && tmp.substr(tmp.size() - 4, tmp.size() - 1) == ".lua")
486  tmp = tmp.substr(0, tmp.size() - 4);
487  else if(strip_ext && tmp.substr(tmp.size() - 5, tmp.size() - 1) == ".lua~")
488  continue;
489  box.addItem(new CEGUI::ListboxTextItem{tmp});
490  }
491  while(FindNextFile(handle, &fdata));
492  FindClose(handle);
493  }
494  else
495  FindClose(handle);
496 #else
497  // TODO: Use dirent on POSIX compliant systems.
498 #endif
