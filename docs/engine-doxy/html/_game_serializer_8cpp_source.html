95

100  // Do not save nodes and lines!

103 
104  entity_name = "entity_" + std::to_string(ent.first);
105  temp_vars.push_back(entity_name);
106 
107  save_entities_.emplace_back(entity_name + " = game.entity.create()");
108  for(tdt::uint i = 0; i < ent.second.size(); ++i)
109  {
110  if(ent.second.test(i) && i < serializers_.size() && serializers_[i])
111  SAVE_COMPONENT(i, ent.first, entity_name);
112  }
113  }
114  for(auto& ent : save_entities_)
115  file_ << ent << "\n";
116  file_ << "\n";
117  for(auto& comp : save_components_)
118  file_ << comp << "\n";
119  file_ << "\n";
120  save_tasks();
121 
122  // Wave system serialization, requires nodes so might be a bit higher,
123  // but just in case is left till the end of the save file.
124  file_ << save_wave_system(game) << "\n" << save_unlocks();

126 
127  file_ << "\n\n-- AUXILIARY VARIABLES TO BE DELETED:\nto_be_deleted = {\n";
128  tdt::uint count{0}; // Saves vertical space.
129  for(const auto& tmp : temp_vars)

132  if(count++ % 10 == 0)
