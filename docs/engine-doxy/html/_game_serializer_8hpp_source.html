367 {
368  auto comp = entities_.get_component<StructureComponent>(id);
369  std::string comm{
370  "game.entity.add_component(" + tbl_name + ", game.enum.component.structure)\n"
371  + "game.grid.set_walk_through(" + tbl_name + ", " + (comp->walk_through ? "true" : "false") + ")\n"
372  + tbl_name + "_residences = { "
373  };
374  std::string set_residents{}; // This will ensure that the grid nodes will have their residents also set.
375 
376  for(tdt::uint i = 0; i < comp->residences.size(); ++i)
377  {
378  comm.append("entity_" + std::to_string(comp->residences[i]) +
379  (i == comp->residences.size() - 1 ? "" : ", "));
380  set_residents.append("game.grid.set_resident(entity_" + std::to_string(comp->residences[i])
381  + ", " + tbl_name + ")\n"
382  + "game.grid.set_free(entity_" + std::to_string(comp->residences[i]) + ", false)\n");
383 
384  // Portal nodes have residents, so this avoids check on free nodes.
385  auto node = entities_.get_component<GridNodeComponent>(comp->residences[i]);
386  if(node && node->neighbours[DIRECTION::PORTAL] != Component::NO_ENTITY)
387  {
388  set_residents.append("game.grid.set_portal_neighbour(entity_" + std::to_string(comp->residences[i])
389  + ", entity_" + std::to_string(node->neighbours[DIRECTION::PORTAL]) + ")\n");
390  }
391  }
392  comm.append(" }\ngame.grid.add_residences(" + tbl_name + ", '" + tbl_name + "_residences')\n");
393  comm.append(tbl_name + "_residences = nil");
394 
395  save_components_.emplace_back(std::move(comm));
396  save_components_.emplace_back(std::move(set_residents));
397 }
