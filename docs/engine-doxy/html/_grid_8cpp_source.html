186 bool Grid::distribute_to_adjacent_free_nodes(EntitySystem& ents, tdt::uint node, const std::vector<tdt::uint>& ids)
187 {
188  if(!GridNodeHelper::is_free(ents, node))
189  return false;
190 
191  std::deque<tdt::uint> queue{};
192  std::set<tdt::uint> visited{};
193  tdt::uint distribution_count{};
194 
195  queue.push_back(node);
196  while(!queue.empty() && distribution_count < ids.size())
197  {
198  auto current = queue.front();
199  queue.pop_front();
200  if(!visited.insert(current).second)
201  continue; // Already visited.
202 
203  // It's free because of check on neighbours and node at the top.
204  PhysicsHelper::set_2d_position(ents, ids[distribution_count++],
205  PhysicsHelper::get_2d_position(ents, current));
206 
207  const auto& neighbours = GridNodeHelper::get_neighbours(ents, current);
208  for(const auto& neighbour : neighbours)
209  {
210  if(visited.count(neighbour) == 0 && GridNodeHelper::is_free(ents, neighbour))
211  queue.push_back(neighbour);
212  }
213  }
214 
215  return distribution_count == ids.size();
216 }
