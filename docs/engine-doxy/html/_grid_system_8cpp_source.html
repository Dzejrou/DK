21  if(graphics_loaded_)
22  {

25 
26  for(const auto& node : freed)
27  GraphicsHelper::set_material(entities_, node, "colour/blue");
28  }
29 
30  std::set<tdt::uint> processed_nodes{}; // Makes sure node aren't processed multiple times.
31  for(const auto& node : unfreed)
32  {
33  update_neighbours_(node); // Updates a block when placed, not needed when freed.
34  for(const auto& neighbour : GridNodeHelper::get_neighbours(entities_, node))
35  {
36  if(neighbour == Component::NO_ENTITY)
37  continue;
38  if(!processed_nodes.insert(neighbour).second)
39  continue;
40  update_neighbours_(neighbour);
41  }
42  }
43  for(const auto& node : freed)
44  {
45  for(const auto& neighbour : GridNodeHelper::get_neighbours(entities_, node))
46  {
47  if(neighbour == Component::NO_ENTITY)
48  continue;
49  if(!processed_nodes.insert(neighbour).second)
50  continue;
51  update_neighbours_(neighbour);
52  }
53  }
54 
55  // Correct pathfinding.
56  if(!unfreed.empty())
57  {
58  for(auto& ent : entities_.get_component_container<PathfindingComponent>())
59  {
60  for(auto node : unfreed)
61  {
62  if(std::find(ent.second.path_queue.begin(), ent.second.path_queue.end(),
63  node) != ent.second.path_queue.end())
64  {
65  if(!util::pathfind<util::DEFAULT_PATHFINDING_ALGORITHM>(
66  entities_, ent.first, ent.second.target_id, util::DEFAULT_HEURISTIC{entities_}, true))
67  { // Can't correct the path.
68  ent.second.path_queue.clear();
69  ent.second.target_id = Component::NO_ENTITY;
70  }
71  break; // Other unfreed nodes were taken into account already.
72  }
73  }
74  }
75  }
76  Grid::instance().clear_unfreed();
77  Grid::instance().clear_freed();
78 }

168 void GridSystem::update_neighbours_(tdt::uint id)
169 {
170  auto comp = entities_.get_component<GridNodeComponent>(id);
171 
172  if(!comp || comp->resident == Component::NO_ENTITY) // Saves 2 component lookups.
173  return;
174 
175  auto node_phys = entities_.get_component<PhysicsComponent>(id);
176  auto graph = entities_.get_component<GraphicsComponent>(comp->resident);
177  auto phys = entities_.get_component<PhysicsComponent>(comp->resident);
178  auto align = entities_.get_component<AlignComponent>(comp->resident);
179  if(comp && graph && phys && graph->node && graph->entity && align && node_phys)
180  {
181 
182  auto& neigh = comp->neighbours;
183  int active_main_neighbours{0}; // Main neighbours: UP, DOWN, LEFT, RIGHT.
184 
185  // Only align to other blocks with align component.
186  bool up = !GridNodeHelper::is_free(entities_, neigh[DIRECTION::UP])
187  && entities_.has_component<AlignComponent>(GridNodeHelper::get_resident(entities_, neigh[DIRECTION::UP]));
188  bool down = !GridNodeHelper::is_free(entities_, neigh[DIRECTION::DOWN])
189  && entities_.has_component<AlignComponent>(GridNodeHelper::get_resident(entities_, neigh[DIRECTION::DOWN]));
190  bool left = !GridNodeHelper::is_free(entities_, neigh[DIRECTION::LEFT])
191  && entities_.has_component<AlignComponent>(GridNodeHelper::get_resident(entities_, neigh[DIRECTION::LEFT]));
192  bool right = !GridNodeHelper::is_free(entities_, neigh[DIRECTION::RIGHT])
193  && entities_.has_component<AlignComponent>(GridNodeHelper::get_resident(entities_, neigh[DIRECTION::RIGHT]));
194  if(up)
195  ++active_main_neighbours;
196  if(down)
197  ++active_main_neighbours;
198  if(left)
199  ++active_main_neighbours;
200  if(right)
201  ++active_main_neighbours;
202 
203  if(up && down || left && right) // This will act like the block is surrounded.
204  active_main_neighbours = 4;
205 
206  graph->material = align->states[active_main_neighbours].material;
207  graph->mesh = align->states[active_main_neighbours].mesh;
208  graph->scale = align->states[active_main_neighbours].scale;
209 
210  graph->node->setScale(graph->scale);
211  graph->node->detachObject(graph->entity);
212  scene_mgr_.destroyEntity(graph->entity);
213  graph->entity = scene_mgr_.createEntity(graph->mesh);
214  graph->node->attachObject(graph->entity);
215  if(graph->material != "NO_MAT")
216  graph->entity->setMaterialName(graph->material);
217  graph->node->setOrientation(Ogre::Quaternion{}); // Reverses any rotations.
218  graph->entity->setRenderingDistance(4000.f);
219 
220  Ogre::Vector3 pos{node_phys->position};
221  if(active_main_neighbours == 1)
222  { // Lean to the neighbour.
223  if(up)
224  {
225  pos.z -= graph->scale.x;
226  graph->node->rotate(Ogre::Vector3{0.f, 1.f, 0.f}, Ogre::Degree{90.f});
227  }
228  else if(down)
229  {
230  pos.z += graph->scale.x;
231  graph->node->rotate(Ogre::Vector3{0.f, 1.f, 0.f}, Ogre::Degree{270.f});
232  }
233  else if(left)
234  pos.x -= graph->scale.z / 2;
235 
236  else if(right)
237  pos.x += graph->scale.z / 2;
238  }
239  else
240  pos += align->states[active_main_neighbours].position_offset;
241  phys->half_height = graph->entity->getWorldBoundingBox(true).getHalfSize().y;
242  phys->position.x = pos.x;
243  phys->position.y = phys->half_height;
244  phys->position.z = pos.z;
245  graph->node->setPosition(phys->position);
246 
247  if(active_main_neighbours == 2)
248  { // Apply correct rotation to the half cube.
249  if(up && left)
250  graph->node->rotate(Ogre::Vector3{0.f, 1.f, 0.f}, Ogre::Degree{90.f});
251  else if(up && right)
252  graph->node->rotate(Ogre::Vector3{0.f, 1.f, 0.f}, Ogre::Degree{0.f});
253  else if(down && left)
254  graph->node->rotate(Ogre::Vector3{0.f, 1.f, 0.f}, Ogre::Degree{180.f});
255  else if(down && right)
256  graph->node->rotate(Ogre::Vector3{0.f, 1.f, 0.f}, Ogre::Degree{270.f});
257  }
258  }
