26 bool OptionsWindow::key_pressed(CEGUI::Key::Scan key)
27 {
28  if(is_visible())
29  {
30  if(currently_binded_action_ >= 1 &&
31  currently_binded_action_ < KEY_BIND_ACTION::COUNT &&
32  key_names_.find(key) != key_names_.end())
33  {
34  key_binds_[currently_binded_action_] = key;
35  currently_binded_action_ = KEY_BIND_ACTION::NONE;
36  update_labels_();
37  return true;
38  }
39  else if(key != CEGUI::Key::Scan::Escape)
40  return false; // Allows to close the button.
41  else
42  return true; // Avoids background key activity.
43  }
44  else
45  {
46  for(tdt::uint i = 0; i < KEY_BIND_ACTION::COUNT; ++i)
47  {
48  if(key_binds_[i] == key)
49  {
50  actions_[i]();
51  return true;
52  }
53  }
54  return false;
55  }

307 void OptionsWindow::apply_()
308 {
309  if(!window_ || !render_window_)
310  return;
311 
312  auto sel_mode = ((CEGUI::Listbox*)window_->getChild("FRAME/WINDOW_MODE_LISTBOX"))->getFirstSelectedItem();
313  if(sel_mode)
314  fullscreen_ = sel_mode->getText() == "FULLSCREEN";
315 
316  auto sel_res = ((CEGUI::Listbox*)window_->getChild("FRAME/RESOLUTION_LISTBOX"))->getFirstSelectedItem();
317  if(sel_res)
318  {
319  for(const auto& r : resolutions_)
320  {
321  if(r.first == sel_res->getText())
322  {
323  width_ = std::get<0>(r.second);
324  height_ = std::get<1>(r.second);
325  }
326  }
327  }
328 
329  render_window_->setFullscreen(fullscreen_, width_, height_);
330  renderer_->setDisplaySize(CEGUI::Sizef{(float)view_->getActualWidth(), (float)view_->getActualHeight()});
331  currently_binded_action_ = KEY_BIND_ACTION::NONE;
332 
336  std::ofstream output{"keybinds.lua"};
337  for(int i = 1; i < KEY_BIND_ACTION::COUNT; ++i)
338  {
339  output << "game.set_key_bind(" << std::to_string(i) << ", "
340  << std::to_string(key_binds_[i]) << ")\n";
341  }
342  output.flush();
343  output.close();
344 
345  update_fonts_();
346 }
