44  static std::deque<tdt::uint> get_path(EntitySystem& ents, tdt::uint id, tdt::uint start, tdt::uint end, util::heuristic::HEURISTIC& heuristic,
45  bool allow_destruction = true)
46  {
47  auto comp = ents.get_component<PathfindingComponent>(id);
48  if(!comp)
49  return std::deque<tdt::uint>{};
50 
51  auto& script = lpp::Script::instance();
52 
53  std::map<tdt::uint, tdt::uint> path_edges{};
54  std::set<tdt::uint> open{start};
55  std::map<tdt::uint, tdt::real> score;
56  std::map<tdt::uint, tdt::real> estimate;
57 
58  for(auto& node : ents.get_component_container<GridNodeComponent>())
59  { // Starting score and estimate is "infinity".
60  score.emplace(node.first, std::numeric_limits<tdt::real>::max());
61  estimate.emplace(node.first, std::numeric_limits<tdt::real>::max());
62  }
63  score[start] = 0;
64  estimate[start] = heuristic.get_cost(start, end);
65 
66  tdt::uint current{};
67  bool found_path{false};
68  while(!open.empty())
69  {
70  // Find the best candidate in the open set. Since open is not empty the
71  // returned iterator will be dereferencable.
72  current = *std::min_element(open.begin(), open.end(),
73  [&estimate](const tdt::uint& lhs, const tdt::uint& rhs) -> bool
74  { return estimate[lhs] < estimate[rhs]; });
75  if(current == end && !found_path)
76  {
77  found_path = true;
78  if(PATH_TYPE::return_path())
79  break;
80  }
81  open.erase(current);
82 
83  auto& neighbours = GridNodeHelper::get_neighbours(ents, current);
84  for(tdt::uint i = 0; i < neighbours.size(); ++i)
85  {
86  auto& neighbour = neighbours[i];
87 
88  bool cannot_pass = !GridNodeHelper::is_free(ents, neighbour) &&
89  (!allow_destruction || !PathfindingHelper::can_break(id, *comp, neighbour))
90  && !StructureHelper::is_walk_through(ents, GridNodeHelper::get_resident(ents, neighbour));
91  if(!Grid::instance().in_board(neighbour) || cannot_pass)
92  continue;
93 
94  tdt::real s = PathfindingHelper::get_cost(id, *comp, current, (DIRECTION::VAL)i);
95  tdt::real h = heuristic.get_cost(neighbour, end);
96  auto new_score = score[current] + s;
97 
98  // Either unvisited or we found a better path to it.
99  if(new_score < score[neighbour])
100  {
101  path_edges[neighbour] = current;
102  score[neighbour] = new_score;
103  estimate[neighbour] = new_score + h;
104 
105  if(found_path && PATH_TYPE::return_path())
106  break;
107 
108  open.insert(neighbour);
109  }
110  }
111  }
112 
113  if(found_path)
114  { // Reconstruct the path.
115  std::deque<tdt::uint> path;
116  current = end;
117  path.push_back(current);
118 
119  // Using assignment to an auxiliary iterator in the condition saves another lookup.
120  auto it = path_edges.begin();
121  while((it = path_edges.find(current)) != path_edges.end())
122  {
123  current = it->second;
124  path.push_front(current);
125  }
126  return path;
127  }
128  else
129  return std::deque<tdt::uint>{};
130  }
131  };
132 }
