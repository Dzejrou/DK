57  if(comp1 && comp2)
58  {

61  return;
62 
63  comp2->source = ent_id;
64 
65  if(priority)

69  comp1->task_queue.push_front(comp1->curr_task);

72  comp1->task_queue.push_front(task_id);

74  }
