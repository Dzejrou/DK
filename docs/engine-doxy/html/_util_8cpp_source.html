54 int util::get_enum_direction(EntitySystem& ents, tdt::uint id, tdt::uint target)
55 {
56  auto pos1 = PhysicsHelper::get_position(ents, id);
57  auto pos2 = PhysicsHelper::get_position(ents, target);
58  auto dist = Grid::instance().get_distance();
59 
60  auto diff_x = pos1.x > pos2.x ? pos1.x - pos2.x : pos2.x - pos1.x;
61  bool same_axis_x = diff_x < dist;
62 
63  auto diff_z = pos1.z > pos2.z ? pos1.z - pos2.z : pos2.z - pos1.z;
64  bool same_axis_z = diff_z < dist;
65 
66  bool left = pos1.x > pos2.x;
67  bool right = pos1.x <= pos2.x;
68  bool up = pos1.z > pos2.z;
69  bool down = pos1.z <= pos2.z;
70 
71  if(same_axis_x && same_axis_z)
72  return DIRECTION::NONE;
73  else if(same_axis_x && up)
74  return DIRECTION::UP;
75  else if(same_axis_x && down)
76  return DIRECTION::DOWN;
77  else if(same_axis_z && left)
78  return DIRECTION::LEFT;
79  else if(same_axis_z && right)
80  return DIRECTION::RIGHT;
81  else if(up && left)
82  return DIRECTION::UP_LEFT;
83  else if(up && right)
84  return DIRECTION::UP_RIGHT;
85  else if(down && left)
86  return DIRECTION::DOWN_LEFT;
87  else if(down && right)
88  return DIRECTION::DOWN_RIGHT;
89  else
90  return DIRECTION::NONE;
91 }
