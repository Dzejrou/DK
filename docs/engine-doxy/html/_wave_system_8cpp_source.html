15 void WaveSystem::update(tdt::real delta)
16 {
17  switch(state_)
18  {
19  case WAVE_STATE::ACTIVE:
20  if(entities_spawned_ < entities_total_)
21  {
22  if(spawn_timer_ >= spawn_cooldown_)
23  {
24  spawn_timer_ = tdt::real{};
25  spawn();
26  update_label_text();
27  }
28  else
29  spawn_timer_ += delta;
30  }
31  break;
32  case WAVE_STATE::WAITING:
33  if(second_timer_ >= 1.f)
34  {
35  second_timer_ = tdt::real{};
36  if(next_wave_countdown_ == 0)
37  start_wave();
38  else
39  {
40  --next_wave_countdown_;
41  update_label_text();
42  }
43  }
44  else
45  second_timer_ += delta;
46  break;
47  case WAVE_STATE::INACTIVE:
48  break;
49  }
50 }

137 void WaveSystem::update_label_text()
138 {
139  if(!window_)
140  return;
141 
142  switch(state_)
143  {
144  case WAVE_STATE::ACTIVE:
145  label_text_ = "WAVE #" + std::to_string(curr_wave_number_ + 1) + ": "
146  + std::to_string(wave_entities_) + " / " + std::to_string(entities_spawned_)
147  + " (" + std::to_string(entities_total_) + ")";
148  break;
149  case WAVE_STATE::WAITING:
150  {
151  tdt::uint cdown = next_wave_countdown_;
152  tdt::uint seconds = cdown % 60;
153  cdown /= 60;
154  tdt::uint minutes = cdown % 60;
155  cdown /= 60;
156  tdt::uint hours = cdown % 60;
157 
158  label_text_ = "NEXT WAVE IN: " + std::to_string(hours) + " - "
159  + std::to_string(minutes) + " - " + std::to_string(seconds);
160  break;
161  }
162  case WAVE_STATE::INACTIVE:
163  label_text_ = "WAVE SYSTEM OFFLINE";
164  break;
165  }
166  window_->setText(label_text_);
167 }

293 void WaveSystem::spawn()
294 {
295  if(entities_spawned_ >= entities_total_)
296  return;
297 
298  tdt::uint to_spawn{entities_total_ - entities_spawned_};
299  if(spawning_nodes_.size() < to_spawn)
300  to_spawn = spawning_nodes_.size();
301 
302  tdt::uint spawned{};
303  for(tdt::uint i = 0; i < to_spawn; ++i)
304  {
305  tdt::uint id{};
306  if(i + entities_spawned_ < entity_blueprints_.size())
307  id = entities_.create_entity(entity_blueprints_[i + entities_spawned_]);
308  else if(0 < entity_blueprints_.size())
309  id = entities_.create_entity(entity_blueprints_[0]);
310  else
311  continue;
312 
313  entities_.add_component<DestructorComponent>(id); // It should have it, but just in case.
314  auto comp = entities_.get_component<DestructorComponent>(id);
315  comp->blueprint = "wave_entity_destructor";
316 
317  PhysicsHelper::set_2d_position(entities_, id, PhysicsHelper::get_2d_position(entities_, spawning_nodes_[i]));
318  ++spawned;
319  }
320 
321  entities_spawned_ += spawned;
322  wave_entities_ += spawned;
323  update_label_text();
324 }
